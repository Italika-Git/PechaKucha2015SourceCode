Cacheo de Objetos JQuery

El objetivo del este ejemplo es demostrar el uso de caché con los objetos creados con JQuery para modificar varias propiedades a la vez. Recuerda que cada llamada a un selector retorna un nuevo objeto lo que incrementa el uso de memoria.

Parrafo sin estilo.

Código fuente:

El ejemplo anterior realiza varias operaciones sobre el mismo nodo <p>, por lo que resulta conveniente realizar el cacheo del nodo en una variable evitando también, repetir código. 

                
(function (Init) {
    // El objeto jQuery se pasa como parametro
    Init(window.jQuery, window, document);
}(function ($, window, document) {
    // DomReady
    $(function () {
        // The DOM is ready!
        /* Forma 1: Sin cache */
        $('#btn_MostrarParrafo').on('click', function () {
            $('#contenedorParrafo p').text(chance.paragraph());
            $('#contenedorParrafo p').css('background-color',chance.color());
            $('#contenedorParrafo p').fadeTo(200,0.7);
            $('#contenedorParrafo p').fadeTo(200,1.0);
        });

        /* Forma 2: Con cache */
        var $p = $('p', '#contenedorParrafo');
        $('#btn_MostrarParrafo').on('click', $p, function () {
            $p.text(chance.paragraph());
            $p.css('background-color',chance.color());
            $p.fadeTo(200,0.7);
            $p.fadeTo(200,1.0);
        });
        
        /* Forma 3: Encadenando metodos (Chaining) */
        var $p = $('p', '#contenedorParrafo');
        $('#btn_MostrarParrafo').on('click', $p, function () {
            $p.text(chance.paragraph())
            .css('background-color',chance.color())
            .fadeTo(200,0.7)
            .fadeTo(200,1.0);
        });
        
    }); //end of DOMReady
}));
